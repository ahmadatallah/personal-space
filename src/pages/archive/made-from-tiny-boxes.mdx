import GalleryWrapper from '../../components/gallery-wrapper';
import { graphql } from 'gatsby';

export const query = graphql`
  query madeFromTinyBoxes {
    images: allFile(
      filter: { relativeDirectory: { eq: "made-from-tiny-boxes" } }
      sort: { fields: name }
    ) {
      edges {
        node {
          childImageSharp {
            thumb: gatsbyImageData(layout: CONSTRAINED, width: 500)
            full: gatsbyImageData(layout: CONSTRAINED, width: 1024)
          }
          name
        }
      }
    }
  }
`;

<GalleryWrapper
  images={props.data.images.edges.map(({ node }) => ({
    ...node.childImageSharp,
    caption: node.name,
    thumbAlt: node.name,
  }))}
/>

Some tiny photos I took from Miniatur Wunderland, Hamburg.
