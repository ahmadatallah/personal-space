import GalleryWrapper from '../../components/gallery/gallery-wrapper';
import { graphql } from 'gatsby';

export const query = graphql`
  query poorEaster {
    images: allFile(
      filter: { relativeDirectory: { eq: "poor-easter" } }
      sort: { fields: name }
    ) {
      edges {
        node {
          childImageSharp {
            thumb: gatsbyImageData(layout: CONSTRAINED, width: 500)
            full: gatsbyImageData(layout: FULL_WIDTH)
          }
          name
        }
      }
    }
  }
`;

<GalleryWrapper
  images={props.data.images.edges.map(({ node }) => ({
    ...node.childImageSharp,
    caption: node.name,
    thumbAlt: node.name,
  }))}
/>
