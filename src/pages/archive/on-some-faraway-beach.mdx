import GalleryWrapper from '../../components/gallery/gallery-wrapper';
import { graphql } from 'gatsby';

export const query = graphql`
  query onsomefarawaybeach {
    images: allFile(
      filter: { relativeDirectory: { eq: "on-some-faraway-beach" } }
      sort: { fields: name }
    ) {
      edges {
        node {
          childImageSharp {
            thumb: gatsbyImageData(layout: CONSTRAINED, width: 450, height: 300)
            full: gatsbyImageData(layout: FULL_WIDTH)
          }
          name
        }
      }
    }
  }
`;

<GalleryWrapper
  images={props.data.images.edges.map(({ node }) => ({
    ...node.childImageSharp,
    caption: node.name,
    thumbAlt: node.name,
  }))}
/>
