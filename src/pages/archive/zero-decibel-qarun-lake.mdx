import GalleryWrapper from '../../components/gallery-wrapper';
import { graphql } from 'gatsby';

export const query = graphql`
  query zeroDecibel {
    images: allFile(
      filter: { relativeDirectory: { eq: "zero-decibel-qarun-lake" } }
      sort: { fields: name }
    ) {
      edges {
        node {
          childImageSharp {
            thumb: gatsbyImageData(layout: CONSTRAINED, width: 270, height: 270)
            full: gatsbyImageData(layout: FULL_WIDTH, width: 1024)
          }
          name
        }
      }
    }
  }
`;

<GalleryWrapper
  images={props.data.images.edges.map(({ node }) => ({
    ...node.childImageSharp,
    caption: node.name,
    thumbAlt: node.name,
  }))}
/>
