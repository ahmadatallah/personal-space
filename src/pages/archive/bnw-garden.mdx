import GalleryWrapper from '../../components/gallery-wrapper';
import { graphql } from 'gatsby';

export const query = graphql`
  query bnwGarden {
    images: allFile(
      filter: { relativeDirectory: { eq: "bnw-garden" } }
      sort: { fields: name }
    ) {
      edges {
        node {
          childImageSharp {
            thumb: gatsbyImageData(layout: CONSTRAINED, width: 500, height: 300)
            full: gatsbyImageData(layout: FULL_WIDTH)
          }
          name
        }
      }
    }
  }
`;

<GalleryWrapper
  images={props.data.images.edges.map(({ node }) => ({
    ...node.childImageSharp,
    caption: node.name,
    thumbAlt: node.name,
  }))}
/>
