import Gallery from '@browniebroke/gatsby-image-gallery';
import { graphql } from 'gatsby';

export const query = graphql`
  query windowxp {
    images: allFile(
      filter: { relativeDirectory: { eq: "windows-xp-palette" } }
      sort: { fields: name }
    ) {
      edges {
        node {
          childImageSharp {
            thumb: fluid(maxWidth: 450, maxHeight: 300) {
              ...GatsbyImageSharpFluid
              originalName
            }
            full: fluid(maxWidth: 1024) {
              ...GatsbyImageSharpFluid
            }
          }
          name
        }
      }
    }
  }
`;

<Gallery
  images={props.data.images.edges.map(({ node }) => ({
    ...node.childImageSharp,
    caption: node.name,
    thumbAlt: node.name,
  }))}
/>

Using Windows XP color palette and [@components_ai](https://components.ai/)
