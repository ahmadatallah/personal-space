import GalleryWrapper from '../../components/gallery/gallery-wrapper';
import { graphql } from 'gatsby';

export const query = graphql`
  query hal9000 {
    images: allFile(
      filter: { relativeDirectory: { eq: "hal9000-gen-art" } }
      sort: { fields: name }
    ) {
      edges {
        node {
          childImageSharp {
            thumb: gatsbyImageData(layout: CONSTRAINED, width: 450)
            full: gatsbyImageData(layout: FULL_WIDTH)
          }
          name
        }
      }
    }
  }
`;

<GalleryWrapper
  images={props.data.images.edges.map(({ node }) => ({
    ...node.childImageSharp,
    caption: node.name,
    thumbAlt: node.name,
  }))}
/>

Generative art for HAL 9000 from 2001: A Space Odyssey by overlaying three radial gradients generated and saved from
[@components_ai](https://components.ai/)
